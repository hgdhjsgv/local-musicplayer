//  打开动画
void MainWindow::showAnimation(QWidget* window){
    QPropertyAnimation animation(window,"pos");
    animation.setDuration(500);
    animation.setStartValue(QPoint(this->width(),0));
    animation.setEndValue(QPoint(this->width()-ui->musicList->width(),0));
    animation.start();
    QEventLoop loop;
    connect(&animation,&QPropertyAnimation::finished,&loop,&QEventLoop::quit);
    loop.exec();
}

//加载指定文件夹
void MainWindow::loadMusicDir(const QString& filepath){
    QDir dir(filepath);
    if(dir.exists() == false){
        QMessageBox::warning(this,"文件夹","文件夹打开失败");
        return ;
    }

    //dir.entryInfoList(QDir::Files);
    QFileInfoList fileList = dir.entryInfoList(QDir::Files);
    for(auto element : fileList){
        if(element.suffix() ==  "mp3"){
            ui->musicList->addItem(element.baseName());
        }
    }
    ui->musicList->setCurrentRow(0);
}

// 获取歌词并解析
QMap<qint64,QString> parseLyrics(const QString& filepath){
    QMap<qint64,QString> lyricsMap;
    QFile file(filepath);
    if(!file.open(QIODevice::ReadOnly | QIODevice::Text)){
        qDebug() << "无法打开歌词文件";
         return lyricsMap;
    }
    QTextStream in(&file);
    //in.setCodec("UTF-8");
    QRegularExpression timeRegex(R"(^\[(\d+):(\d+)\.(\d+)\](.*))");
    while(!in.atEnd()){
        QString  line=in.readLine();
        //qDebug()<<line;
        QRegularExpressionMatch match=timeRegex.match(line);
        if(match.hasMatch()){
            int minutes=match.captured(1).toInt();
            int seconds=match.captured(2).toInt();
            int milliseconds=match.captured(3).toInt();
            qint64 timestamp=minutes*60000+seconds*1000+milliseconds;
            QString text=match.captured(4).trimmed();
            lyricsMap.insert(timestamp,text);
        }
    }
    file.close();
    return lyricsMap;
}

// 加载行数
void MainWindow::updateLyrics(QMap<qint64,QString>& lyricsMap) {
    // 找到当前应该显示的歌词行
    currentTime = m_player->position();
    auto it = lyricsMap.upperBound(currentTime); // 找到第一个不小于 currentTime 的时间戳

    ui->lyricslist->clear();

    // 显示当前行及其前后几行
    int count = 0;
    const int displayLines = 5; // 显示5行歌词

    // 回退显示前几行（如果有）
    auto prevIt = it;
    for (int i = 0; i < 2 && prevIt != lyricsMap.begin(); ++i) {
        --prevIt;
    }

    // 显示前几行到当前行
    for (auto currentIt = prevIt; currentIt != lyricsMap.end() && count < displayLines; ++currentIt, ++count) {
        QListWidgetItem *item = new QListWidgetItem(currentIt.value());
        auto nextIt=currentIt;
        nextIt++;
        if (currentIt.key() == currentTime ||
            (it != lyricsMap.end() && currentIt.key() ==it.key()) ) {
            // 当前行或下一行（处理边界情况）
            item->setForeground(Qt::red);
        } else if (currentIt.key() < currentTime &&
                   (nextIt == lyricsMap.end() || nextIt.key() > currentTime)) {
            // 上一行（如果当前时间已经过了该行但还没到下一行）
            item->setForeground(Qt::blue);
            QFont font = item->font();
            font.setFamily("Arial");
            font.setPointSize(16);
            font.setBold(true);
            item->setFont(font);
        } else {
            item->setForeground(Qt::black);
        }
        ui->lyricslist->addItem(item);
    }
    QListWidgetItem *item= new QListWidgetItem(prevIt.value());
    // 滚动到当前行（如果存在）
    if (!lyricsMap.isEmpty() && it != lyricsMap.end()) {
        int rowIndex = 0;
        for (auto currentIt = prevIt; currentIt != it; ++currentIt, ++rowIndex) {
            if (currentIt == lyricsMap.end()) break;
        }
        if (rowIndex < ui->lyricslist->count()) {
            ui->lyricslist->scrollToItem(item, QAbstractItemView::PositionAtCenter);
        }
    }
}

// 找到当前进度对应的歌词
void MainWindow::setCurrentTime() {
    QMap lyMap=parseLyrics(musicLy);
    updateLyrics(lyMap);
}
